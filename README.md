function daysBetween(start, end) {
  const s = new Date(start);
  const e = new Date(end);
  const diff = Math.ceil((e - s) / (1000 * 60 * 60 * 24));
  return Math.max(diff, 0);
}

function overlaps(aStart, aEnd, bStart, bEnd) {
  return new Date(aStart) < new Date(bEnd) && new Date(bStart) < new Date(aEnd);
}

function formatEuro(value) {
  return new Intl.NumberFormat("it-IT", { style: "currency", currency: "EUR" }).format(Number(value || 0));
}

function formatDate(date) {
  if (!date) return "-";
  return new Intl.DateTimeFormat("it-IT").format(new Date(date));
}

function todayIso() {
  return new Date().toISOString().slice(0, 10);
}

function buildMonthDays(baseDate) {
  const year = baseDate.getFullYear();
  const month = baseDate.getMonth();
  const daysInMonth = new Date(year, month + 1, 0).getDate();
  return Array.from({ length: daysInMonth }, (_, i) => new Date(year, month, i + 1));
}

    { name: "overlaps true", ok: overlaps("2026-04-01", "2026-04-03", "2026-04-02", "2026-04-05") === true },
    { name: "overlaps false on checkout boundary", ok: overlaps("2026-04-01", "2026-04-03", "2026-04-03", "2026-04-05") === false },
  ];

  const failed = cases.filter((test) => !test.ok);
  if (failed.length > 0) {
    console.error("Utility tests failed:", failed.map((t) => t.name).join(", "));
  }
}

                      <div className="relative">
                        <Search className="absolute left-3 top-1/2 h-4 w-4 -translate-y-1/2 text-slate-400" />
                        <Input className="w-full rounded-2xl pl-9 md:w-72" placeholder="Cerca ospite, camera, telefono, canale" value={search} onChange={(e) => setSearch(e.target.value)} />
                      </div>
                      <Select value={statusFilter} onValueChange={setStatusFilter}>
                        <SelectTrigger className="w-full rounded-2xl md:w-48"><SelectValue /></SelectTrigger>
                        <SelectContent>
                          <SelectItem value="Tutti">Tutti gli stati</SelectItem>
                          {statusOptions.map((status) => <SelectItem key={status} value={status}>{status}</SelectItem>)}
                        </SelectContent>
                      </Select>
                      <Select value={channelFilter} onValueChange={setChannelFilter}>
                        <SelectTrigger className="w-full rounded-2xl md:w-56"><SelectValue /></SelectTrigger>
                        <SelectContent>
                          <SelectItem value="Tutti">Tutti i canali</SelectItem>
                          {channelOptions.map((channel) => <SelectItem key={channel} value={channel}>{channel}</SelectItem>)}
                        </SelectContent>
                      </Select>
                    </div>
                  </CardHeader>
                  <CardContent className="space-y-4">
                    {filteredBookings.length === 0 ? (
                      <div className="rounded-2xl border border-dashed border-slate-300 p-8 text-center text-slate-500">Nessuna prenotazione trovata.</div>
                    ) : filteredBookings.map((booking) => {
                      const room = roomMap[booking.roomId];
                      return (
                        <div key={booking.id} className="flex flex-col gap-4 rounded-2xl border border-slate-200 bg-white p-4">
                          <div className="flex flex-col gap-4 md:flex-row md:items-start md:justify-between">
                            <div className="space-y-2">
                              <div className="flex flex-wrap items-center gap-2">
                                <h3 className="text-lg font-semibold text-slate-900">{booking.guest}</h3>
                                <Badge variant="secondary" className="rounded-xl">{booking.status}</Badge>
                                <Badge variant="outline" className="rounded-xl">{booking.channel}</Badge>
                              </div>
                              <p className="text-sm text-slate-600">{room?.name} · {booking.guests} ospiti · {formatDate(booking.checkIn)} → {formatDate(booking.checkOut)}</p>
                              <p className="text-sm text-slate-500">Data prenotazione: {formatDate(booking.bookingDate)} · Pagamento: {booking.paymentType}</p>
                              <p className="text-sm text-slate-500">{booking.phone || "Nessun telefono"}</p>
                              {booking.notes && <p className="text-sm text-slate-500">Note: {booking.notes}</p>}
                            </div>
                            <div className="grid grid-cols-1 gap-3 md:grid-cols-3">
                              <div className="rounded-2xl bg-slate-50 p-3 text-center"><p className="text-xs text-slate-500">Totale</p><p className="text-lg font-bold text-slate-900">{formatEuro(booking.total)}</p></div>
                              <div className="rounded-2xl bg-slate-50 p-3 text-center"><p className="text-xs text-slate-500">Acconto</p><p className="text-lg font-bold text-slate-900">{formatEuro(booking.deposit)}</p></div>
                              <div className="rounded-2xl bg-slate-50 p-3 text-center"><p className="text-xs text-slate-500">Saldo</p><p className="text-lg font-bold text-slate-900">{formatEuro(booking.balance)}</p></div>
                            </div>
                          </div>
                          <div className="flex flex-wrap justify-end gap-2">
                            <Button variant="outline" className="rounded-2xl" onClick={() => openEditBooking(booking)}><Pencil className="mr-2 h-4 w-4" />Modifica</Button>
                            <Button variant="outline" className="rounded-2xl" onClick={() => removeBooking(booking.id)}><Trash2 className="mr-2 h-4 w-4" />Elimina</Button>
                          </div>
                        </div>
                      );
                    })}
                  </CardContent>
                </Card>
              </TabsContent>

              <TabsContent value="camere">
                <div className="grid grid-cols-1 gap-6 xl:grid-cols-3">
                  <Card className="rounded-2xl border-0 shadow-sm xl:col-span-2">
                    <CardHeader><CardTitle>Camere e disponibilità</CardTitle></CardHeader>
                    <CardContent className="grid grid-cols-1 gap-4 md:grid-cols-2">
                      {availability.map((room) => (
                        <div key={room.id} className="rounded-2xl border border-slate-200 p-4">
                          <div className="flex items-start justify-between gap-3">
                            <div>
                              <h3 className="font-semibold text-slate-900">{room.name}</h3>
                              <p className="text-sm text-slate-500">{room.type} · max {room.capacity} ospiti</p>
                              <p className="mt-1 text-sm font-medium text-slate-700">{formatEuro(room.price)}/notte</p>
                              {room.notes && <p className="mt-1 text-sm text-slate-500">{room.notes}</p>}
                              {room.guest && <p className="mt-2 text-sm text-slate-600">Occupata da: {room.guest}</p>}
                            </div>
                            <Badge className="rounded-xl" variant={room.state === "Disponibile" ? "default" : "secondary"}>{room.state}</Badge>
                          </div>
                          <div className="mt-4 flex gap-2">
                            <Button variant="outline" className="rounded-2xl" onClick={() => openEditRoom(room)}><Pencil className="mr-2 h-4 w-4" />Modifica</Button>
                            <Button variant="outline" className="rounded-2xl" onClick={() => deleteRoom(room.id)}><Trash2 className="mr-2 h-4 w-4" />Elimina</Button>
                          </div>
                        </div>
                      ))}
                    </CardContent>
                  </Card>
                  <Card className="rounded-2xl border-0 shadow-sm">
                    <CardHeader><CardTitle>Funzioni incluse</CardTitle></CardHeader>
                    <CardContent className="space-y-3 text-sm text-slate-600">
                      <div className="flex items-start gap-2"><CheckCircle2 className="mt-0.5 h-4 w-4" /><span>Database online con Supabase</span></div>
                      <div className="flex items-start gap-2"><CheckCircle2 className="mt-0.5 h-4 w-4" /><span>Login amministratore con email e password</span></div>
                      <div className="flex items-start gap-2"><CheckCircle2 className="mt-0.5 h-4 w-4" /><span>CRUD camere e prenotazioni</span></div>
                      <div className="flex items-start gap-2"><CheckCircle2 className="mt-0.5 h-4 w-4" /><span>Acconto, saldo e totale automatici</span></div>
                      <div className="flex items-start gap-2"><CheckCircle2 className="mt-0.5 h-4 w-4" /><span>Calendario mensile camere occupate</span></div>
                    </CardContent>
                  </Card>
                </div>
              </TabsContent>

              <TabsContent value="calendario">
                <Card className="rounded-2xl border-0 shadow-sm">
                  <CardHeader className="flex flex-col gap-4 md:flex-row md:items-center md:justify-between">
                    <CardTitle>Calendario camere occupate</CardTitle>
                    <div className="flex items-center gap-2">
                      <Button variant="outline" className="rounded-2xl" onClick={() => setCalendarDate(new Date(calendarDate.getFullYear(), calendarDate.getMonth() - 1, 1))}>Mese precedente</Button>
                      <div className="min-w-44 text-center text-sm font-medium capitalize text-slate-700">{monthLabel}</div>
                      <Button variant="outline" className="rounded-2xl" onClick={() => setCalendarDate(new Date(calendarDate.getFullYear(), calendarDate.getMonth() + 1, 1))}>Mese successivo</Button>
                    </div>
                  </CardHeader>
                  <CardContent>
                    <div className="overflow-x-auto rounded-2xl border border-slate-200">
                      <div className="min-w-[1200px]">
                        <div className="grid" style={{ gridTemplateColumns: `220px repeat(${monthDays.length}, minmax(38px, 1fr))` }}>
                          <div className="sticky left-0 z-10 border-b border-r bg-slate-100 p-3 text-sm font-semibold text-slate-700">Camera</div>
                          {monthDays.map((day) => <div key={day.toISOString()} className="border-b border-r bg-slate-100 p-2 text-center text-xs font-semibold text-slate-700">{day.getDate()}</div>)}
                          {calendarRows.map(({ room, occupancy }) => (
                            <React.Fragment key={room.id}>
                              <div className="sticky left-0 z-10 border-b border-r bg-white p-3 text-sm font-medium text-slate-800"><div>{room.name}</div><div className="text-xs text-slate-500">{room.type}</div></div>
                              {occupancy.map((booking, idx) => (
                                <div key={`${room.id}-${idx}`} className={`border-b border-r p-1 text-center text-[10px] ${booking ? "bg-slate-800 text-white" : "bg-white text-slate-300"}`} title={booking ? `${booking.guest} · ${formatDate(booking.checkIn)} → ${formatDate(booking.checkOut)}` : "Libera"}>
                                  {booking ? "Occupata" : "·"}
                                </div>
                              ))}
                            </React.Fragment>
                          ))}
                        </div>
                      </div>
                    </div>
                  </CardContent>
                </Card>
              </TabsContent>
            </Tabs>
          </>
        )}

        <Dialog open={bookingOpen} onOpenChange={(value) => { setBookingOpen(value); if (!value) resetBookingForm(); }}>
          <DialogContent className="max-h-[90vh] max-w-3xl overflow-y-auto rounded-2xl">
            <DialogHeader><DialogTitle>{editingBookingId ? "Modifica prenotazione" : "Inserisci prenotazione"}</DialogTitle></DialogHeader>
            <div className="grid grid-cols-1 gap-4 md:grid-cols-2">
              <div className="space-y-2"><Label>Nome ospite *</Label><Input value={bookingForm.guest} onChange={(e) => setBookingForm({ ...bookingForm, guest: e.target.value })} /></div>
              <div className="space-y-2"><Label>Telefono</Label><Input value={bookingForm.phone} onChange={(e) => setBookingForm({ ...bookingForm, phone: e.target.value })} /></div>
              <div className="space-y-2"><Label>Data prenotazione *</Label><Input type="date" value={bookingForm.bookingDate} onChange={(e) => setBookingForm({ ...bookingForm, bookingDate: e.target.value })} /></div>
              <div className="space-y-2"><Label>Camera *</Label><Select value={String(bookingForm.roomId)} onValueChange={(value) => setBookingForm({ ...bookingForm, roomId: value })}><SelectTrigger><SelectValue placeholder="Seleziona una camera" /></SelectTrigger><SelectContent>{rooms.map((room) => <SelectItem key={room.id} value={String(room.id)}>{room.name} · {formatEuro(room.price)}/notte</SelectItem>)}</SelectContent></Select></div>
              <div className="space-y-2"><Label>Check-in *</Label><Input type="date" value={bookingForm.checkIn} onChange={(e) => setBookingForm({ ...bookingForm, checkIn: e.target.value })} /></div>
              <div className="space-y-2"><Label>Check-out *</Label><Input type="date" value={bookingForm.checkOut} onChange={(e) => setBookingForm({ ...bookingForm, checkOut: e.target.value })} /></div>
              <div className="space-y-2"><Label>Numero ospiti</Label><Input type="number" min="1" value={bookingForm.guests} onChange={(e) => setBookingForm({ ...bookingForm, guests: e.target.value })} /></div>
              <div className="space-y-2"><Label>Stato</Label><Select value={bookingForm.status} onValueChange={(value) => setBookingForm({ ...bookingForm, status: value })}><SelectTrigger><SelectValue /></SelectTrigger><SelectContent>{statusOptions.map((status) => <SelectItem key={status} value={status}>{status}</SelectItem>)}</SelectContent></Select></div>
              <div className="space-y-2"><Label>Tipo pagamento</Label><Select value={bookingForm.paymentType} onValueChange={(value) => setBookingForm({ ...bookingForm, paymentType: value })}><SelectTrigger><SelectValue /></SelectTrigger><SelectContent>{paymentOptions.map((item) => <SelectItem key={item} value={item}>{item}</SelectItem>)}</SelectContent></Select></div>
              <div className="space-y-2"><Label>Canale</Label><Select value={bookingForm.channel} onValueChange={(value) => setBookingForm({ ...bookingForm, channel: value })}><SelectTrigger><SelectValue /></SelectTrigger><SelectContent>{channelOptions.map((item) => <SelectItem key={item} value={item}>{item}</SelectItem>)}</SelectContent></Select></div>
              <div className="space-y-2 md:col-span-2"><Label>Acconto versato</Label><Input type="number" min="0" value={bookingForm.deposit} onChange={(e) => setBookingForm({ ...bookingForm, deposit: e.target.value })} /></div>
              <div className="space-y-2 md:col-span-2"><Label>Note</Label><Textarea value={bookingForm.notes} onChange={(e) => setBookingForm({ ...bookingForm, notes: e.target.value })} /></div>
            </div>
            {bookingForm.roomId && bookingForm.checkIn && bookingForm.checkOut && (
              <div className="mt-4 grid grid-cols-1 gap-3 rounded-2xl bg-slate-50 p-4 md:grid-cols-3">
                <div><p className="text-xs text-slate-500">Totale prenotazione</p><p className="text-lg font-bold text-slate-900">{formatEuro(daysBetween(bookingForm.checkIn, bookingForm.checkOut) * Number(roomMap[Number(bookingForm.roomId)]?.price || 0))}</p></div>
                <div><p className="text-xs text-slate-500">Acconto</p><p className="text-lg font-bold text-slate-900">{formatEuro(bookingForm.deposit)}</p></div>
                <div><p className="text-xs text-slate-500">Saldo residuo</p><p className="text-lg font-bold text-slate-900">{formatEuro(Math.max(daysBetween(bookingForm.checkIn, bookingForm.checkOut) * Number(roomMap[Number(bookingForm.roomId)]?.price || 0) - Number(bookingForm.deposit || 0), 0))}</p></div>
              </div>
            )}
            {bookingError && <div className="mt-4 rounded-xl border border-red-200 bg-red-50 p-3 text-sm text-red-700">{bookingError}</div>}
            <div className="mt-6 flex justify-end gap-2"><Button variant="outline" onClick={() => setBookingOpen(false)} className="rounded-2xl">Annulla</Button><Button onClick={saveBooking} className="rounded-2xl">{editingBookingId ? "Salva modifiche" : "Salva prenotazione"}</Button></div>
          </DialogContent>
        </Dialog>

        <Dialog open={roomOpen} onOpenChange={(value) => { setRoomOpen(value); if (!value) resetRoomForm(); }}>
          <DialogContent className="max-w-2xl rounded-2xl">
            <DialogHeader><DialogTitle>{editingRoomId ? "Modifica camera" : "Inserisci camera"}</DialogTitle></DialogHeader>
            <div className="grid grid-cols-1 gap-4 md:grid-cols-2">
              <div className="space-y-2"><Label>Nome camera *</Label><Input value={roomForm.name} onChange={(e) => setRoomForm({ ...roomForm, name: e.target.value })} /></div>
              <div className="space-y-2"><Label>Tipo *</Label><Select value={roomForm.type} onValueChange={(value) => setRoomForm({ ...roomForm, type: value })}><SelectTrigger><SelectValue /></SelectTrigger><SelectContent>{roomTypeOptions.map((item) => <SelectItem key={item} value={item}>{item}</SelectItem>)}</SelectContent></Select></div>
              <div className="space-y-2"><Label>Prezzo a notte *</Label><Input type="number" min="1" value={roomForm.price} onChange={(e) => setRoomForm({ ...roomForm, price: e.target.value })} /></div>
              <div className="space-y-2"><Label>Capienza massima *</Label><Input type="number" min="1" value={roomForm.capacity} onChange={(e) => setRoomForm({ ...roomForm, capacity: e.target.value })} /></div>
              <div className="space-y-2 md:col-span-2"><Label>Note</Label><Textarea value={roomForm.notes} onChange={(e) => setRoomForm({ ...roomForm, notes: e.target.value })} /></div>
            </div>
            {roomError && <div className="mt-4 rounded-xl border border-red-200 bg-red-50 p-3 text-sm text-red-700">{roomError}</div>}
            <div className="mt-6 flex justify-end gap-2"><Button variant="outline" onClick={() => setRoomOpen(false)} className="rounded-2xl">Annulla</Button><Button onClick={saveRoom} className="rounded-2xl">{editingRoomId ? "Salva modifiche" : "Salva camera"}</Button></div>
          </DialogContent>
        </Dialog>
      </div>
    </div>
  );
}
